<script type = "text/javascript">
    'use strict';

    (function($) {
        $.fn.goTo = function() {
            $('html, body').animate({
                scrollTop: $(this).offset().top + 1 + 'px'
            }, 600);
            return this;
        };

        $.fn.bringForward = function() {
            $(this).css('z-index', 1);
        };

        $.fn.sendBack = function() {
            $(this).css('z-index', -1);
        };
    })(jQuery);

   $(document).ready(function() {
       $('.navbar-brand').hide();
       $('#navbar-cesium-com-link').hide();

       $('#exit-btn').click(closeDemo);

       var playButton = $('#play-btn').click(openDemo);
       if (typeof window.WebGLRenderingContext === 'undefined') {
           playButton.hide();
       }

       $('.homepage-header-image, .homepage-header-demo').height($('.homepage-header').height());
   });

   $(window).resize(function () {
       $('.homepage-header-image, .homepage-header-demo').height($('.homepage-header').height());
   });

   var firstTime = true;
   function openDemo() {
       if (firstTime) {
           firstTime = false;
           $('#cesium-view').attr('src', "//cesiumjs.org/Cesium/Build/Apps/CesiumViewer/index.html");
       }

       $('body').goTo();
       $('body').addClass('no-scroll');

       $('#cesium-view').bringForward();

       $('.homepage-header').animate({
           opacity: 0,
           top: -500
       }, 600);

       $('#cesium-com-referral').slideUp();
       $('#exit-btn').show();
   }

   function closeDemo() {
       $('body').removeClass('no-scroll');

       $('#cesium-view').sendBack();

       $('.homepage-header').animate({
           opacity: 100,
           top: 0
       }, 600);

       $('#cesium-com-referral').slideDown();
       $('#exit-btn').hide();
   }
</script>
