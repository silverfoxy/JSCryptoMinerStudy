$(function()
{
	_date = moment().format("MMMM  YYYY");
	$("#cluster-month-date").empty().html(" - "+ _date );

    $.getJSON("http://apps.southsudanngoforum.org/data/cluster_meetings/1", function(res) 
    {
        result = JSON.parse(JSON.stringify(res));
        events = [];
        $.each(result, function(k, v) 
        {
            e ="<div class='forum-calendar-events'><h5><a href='"+ v.link +"'>"+ v.title +"</a></h5><span class='location'>"+ v.location +"</span>";
        	$.each(v.description, function(_k, _v) 
        	{
        		e +="<span class='event-date'>"+ _v +"</span>";
        	});
        	e +="</div>";
            events.push( e );
        });
        $("#cluster-calendar").empty().html( events.join("") );
    });
});
