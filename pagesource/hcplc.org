    <script>
	(function () {
		var m = document.getElementById('multisrch'),
			t = m.querySelectorAll('.srchtabs li'),
			f = m.querySelectorAll('form'),
			cst = m.querySelector('#catalogSearchType'),
			cso = m.querySelector('#catalogSearchOptions'),
			li = cso.querySelectorAll('li'),
			i;
		function doSearch(e) {
			e = e || window.event;
			e.preventDefault();
			var st = m.querySelector('li.active').getAttribute('data-searchtype') || "catalog",
				kw = m.querySelector('#' +st + 'Input').value || "",
				URL = "";
			if (kw.trim() == '') {
				m.querySelector('#' +st + 'Input').focus();
				return false;
			}
			if (ga && ga.hasOwnProperty('loaded') && ga.loaded === true) {
            	ga('send', 'event', {eventCategory: 'search', eventAction: 'submit', eventLabel: st});
			}
			if (st==="ebsco") {
				document.getElementById('ebscoSearch').submit();
			} else if (st==="site") {
				document.getElementById('siteSearch').submit();
			} else {//
				kw = encodeURIComponent(kw);
				var cp = cst.getAttribute('data-value') || "KW";
				URL = "http://catalog.hcplc.org/polaris/Search/searchresults.aspx?type=Keyword&Page=0&Sort=Relevance&by=" + cp + "&term=" + kw;
				window.location = URL;
			}
			return false;
		}
		m.className = 'multisrch-container'; //removes hide-noscript
		for (i=0; i<t.length; i++){
			t[i].addEventListener("click", function(){
				var st = this.getAttribute('data-searchtype'),
					tb = m.querySelectorAll('.srchtabs li'),
					fm = m.querySelectorAll('form');
				for (var j=0; j<tb.length; j++) {
					tb[j].className = '';
					fm[j].className = 'multisrch-form';
					fm[j].setAttribute("aria-hidden","true");
				}
				this.className = 'active';
				m.querySelector('#' + st + 'Search').className += ' active';
				m.querySelector('#' + st + 'Search').setAttribute("aria-hidden","false");
				m.querySelector('#' + st + 'Input').focus();
			})
		}
		for (i=0; i<f.length; i++){f[i].addEventListener("submit", doSearch)}
		cst.addEventListener("click", function(){
			cso.className = (cso.className.match(/cso-open/)) ? "" : "cso-open";
		})
		for (i=0; i<li.length; i++){
			li[i].addEventListener("click", function() {
				for (var j=0; j<li.length; j++) li[j].className="";
				this.className = "active";
				var s = cst.querySelector('span');
				s.setAttribute('data-value', this.getAttribute('data-value'));
				s.innerHTML = this.innerHTML;
				cso.className="";
			})
		}
	})()
