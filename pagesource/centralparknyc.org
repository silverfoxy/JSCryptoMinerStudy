(function($){

  function toggleNav() {
    if ($('#offcanvas-wrapper').hasClass('show-nav')) {
      // Do things on Nav Close
      $('#offcanvas-wrapper').removeClass('show-nav');
      $('#offcanvas-menu').removeClass('offcanvas-menu-active');
      $('#contentDiv a').unbind('click');
    } else {
      // Do things on Nav Open
      $('#offcanvas-wrapper').addClass('show-nav');
      $('#offcanvas-menu').addClass('offcanvas-menu-active');
      $('#contentDiv a').click(function (e){
        e.preventDefault();
      });
    }
  }

  function searchClick() {
    $('.search-bar').toggle(300).toggleClass('search-bar-active');
  }

  $('.alert-btn').on('click', function(){
    $('.alert').slideToggle();
  });

  $('.search-hp').on("click", function(){
    searchClick();
    var searchClickHP = $('.search-bar').hasClass("search-bar-active");
    if (searchClickHP) {
      $('.fa-search.desktop-menu').css("color", "#939493");
      $('#search-form .search-input').focus();
    }
    else {
      $('.fa-search.desktop-menu').css("color", "white");
    }
  });

  $('.search-mobile').on("click", function(){
    searchClick();
    var searchClickHP = $('.search-bar').hasClass("search-bar-active");
    if (searchClickHP) {
      $('.navbar-mobile .fa-search').css("color", "#939493");
      $('.icon-li.search-mobile').css("color", "#939493");
      $('#search-form .search-input').focus();
    }
    else {
      $('.navbar-mobile .fa-search').css("color", "#636466");
      $('.icon-li.search-mobile').css("color", "#636466");
    }
  });

function submitEmail(thisDivID) {
(function($){
  var myEmail = $('#'+thisDivID+' form input.primaryEmail').val();
  $('input.primaryEmail').val(myEmail);
  var myForm = $('#'+thisDivID+' form');
  var myData = 'method=submitSurvey&survey_id=2621';
  luminateExtend.api({
    api: 'survey',
    requestType: 'POST',
    data: myData,
    form: myForm,
    requiresAuth: 'true',
    callback: function(data){
      var successVar = (data.submitSurveyResponse) ? data.submitSurveyResponse.success : '';
      var errorVar = (data.errorResponse) ? data.errorResponse.code : '';
      if(successVar == 'true') {
        var submitOpt = 'email_signup_'+thisDivID;
        submitOptimizelyEvent(submitOpt);
        $('.primaryEmail').hide();
        $('.submitButton').hide();
        $('.noteBox').show().html('<div id="thanksBox">Thank you! To complete your profile, visit our <a href="http://www.centralparknyc.org/cpc-registration.html?show=all" onClick="dataLayer.push({\'eventCategory\' : \'Link\', \'eventAction\' : \'Click\', \'eventLabel\' : \'Tell Us More - EmailWide\', \'eventValue\' : undefined, \'nonInteraction\': false, \'event\' : \'GAEvent\'});">Registration Page.</a></div>');
        dataLayer.push({
          'eventCategory' : 'Email Sign-up',
          'eventAction' : 'Email Sign-up',
          'eventLabel' : thisDivID,
          'eventValue' : undefined,
          'nonInteraction': false,
          'event' : 'GAEvent'
        });
      } else {
        if (errorVar == '1734') {
          var s8 = $('#s8').text();
          $('.noteBox').show().html('<p style="color: #F00">This email address already exists; please <a href="http://support.centralparknyc.org/site/UserLogin?NEXTURL='+s8+'">log in</a> to continue.</p>');
          dataLayer.push({
            'eventCategory' : 'Email Sign-up',
            'eventAction' : 'Duplicate Sign-up',
            'eventLabel' : thisDivID,
            'eventValue' : undefined,
            'nonInteraction': false,
            'event' : 'GAEvent'
          });
        } else {
          $('.noteBox').show().html('<p style="color: #F00">'+data.submitSurveyResponse.errors.errorMessage+'</p>');
          dataLayer.push({
            'eventCategory' : 'Email Sign-up',
            'eventAction' : 'Failed Sign-up',
            'eventLabel' : thisDivID,
            'eventValue' : undefined,
            'nonInteraction': false,
            'event' : 'GAEvent'
          });
        }
      }
    }
  });
})(jQuery);
}
