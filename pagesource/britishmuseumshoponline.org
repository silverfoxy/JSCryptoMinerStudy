!function (global) {

  /**
   * Define the adobe digitalData module and return it to the global namespace

   */
  function defineModule($, Tracking) {

    var digitalData = {};


    var adobeTracking = {
	    _init: function() {
		    this._domInit();
		    this._trackConfirmAndPay();
		    this._trackOrder();
		    this._trackPageType();
		    this._trackPageNameAndChannel();
		    this._trackVisitor();

		    this._trackPDPImpression();
		    this._trackBasketImpression();
		    
			return digitalData;			
	    },
		    

	    _domInit: function() {   
			// Save a copy of`this` that can be passed into event handlers
	   		var _this = this;
      
			// Store session, page type and workflow step information so it can be used in different methods
			this.step = Tracking.Step();
		    this.pageType = Tracking.pageType();
		    this.ses = Tracking.Ses();
		    this.chkStep = Tracking.CheckoutStep();
		    
		    
		    /**
		     * Track social link clicks 
		     */			
			$('.js-socialLink').click(function() {
				var socialNetwork = $(this).attr('title');
				
				global.digitalData.socialLink = socialNetwork;
				//Fire tracking
				_satellite.track('sociallink');
			});


		    /**
		     * Track social share clicks 
		     */			
			$(document).on('click', 'li[data-socialshare]', function() {
				var socialNetwork;
				
				global.digitalData.socialShare = socialNetwork;
			});

			
			/**
			* Track Add to Cart
			* jquery custom event minicart-items-added is set in minicartdetail.js
			*/
			$('body').on('minicart-items-added', function () {
				var prod = $('#tag-invtref').text();
				
				global.digitalData.prod = ';'+prod;
				global.digitalData.event_track = 'scAdd';
				
				//Fire tracking
				_satellite.track('add_to_basket');
			});


			/**
			* Track remove from Cart
			*/
			$('body').on('click', '.js-removeItem', function () {
			    var prod = $(this).closest('tr.prod').data('basket-itemsku');
			    
			    
				global.digitalData.prod = ';'+prod;
				global.digitalData.event_track = 'scRemove';
				
				//Fire tracking
				_satellite.track('cart_removal');
			});


			/**
			* Track Checkout Securely now button click. i.e. starting the checkout process
			*/
			$('body').on('click', '#goToCheckoutmd, .js-startCheckoutBtn, #checkout', function () {
				var pageType, $dataAttributesSelector, prods;
				
				pageType = _this.step.step;
				
				if ( pageType === 'shopcart/shopcart' ) {
					$dataAttributesSelector = $('tr.prod');
				}
				else {
					$dataAttributesSelector = $('#minicartDetailWrapper #minicart_products > li');
				}
				
				prods = [];
				
				$dataAttributesSelector.each(function(){
					prods.push( ';' + $(this).data('basket-itemsku') ); 
				});
				
				prods = prods.join(',');
				
				global.digitalData.prod = prods;
				global.digitalData.event_track = 'scCheckout';
				//Fire tracking
				_satellite.track('checkout_start');				
			});


			/**
			* Ticket book now tracking
			*/
			var tixPageName;
			// Landing page
			$( document ).on( 'ticket-book-now', function () {
  			tixPageName = global.digitalData.pageName;
				global.digitalData.prod = '';
				global.digitalData.pageName = tixPageName + ':landing page';
				//Fire tracking
				_satellite.track('landingPage');
			});
      // Select ticket
			$( document ).on( 'ticket-select-ticket', function () {
				global.digitalData.pageName = tixPageName + ':Select ticket';
				//Fire tracking
				_satellite.track('selectTickets');
			});
      // Choose date and time
			$( document ).on( 'ticket-show-cal', function () {
				global.digitalData.pageName = tixPageName + ':Choose date and time';
				//Fire tracking
				_satellite.track('selectDateTime');
			});      
      // Confirm
			$( document ).on( 'ticket-confirm', function () {
				global.digitalData.pageName = tixPageName + ':Confirm';
				//Fire tracking
				_satellite.track('confirmPage');
			});      
      // Add to basket
			$( '#tickets-catalogue-no, #tickets-catalogue-yes' ).on( 'click', function () {
				var prod = $('#tag-invtref').text();
				
				// Add catalogue prod to tracking too
				if ( $( this ).is( '#tickets-catalogue-yes' ) ) {
  				prod += ';' + Venda.Platform.getUrlParam( $( this ).attr( 'href' ), 'invt' );
				}
				
				global.digitalData.prod = ';'+prod;
				global.digitalData.event_track = 'scAdd';
				
				//Fire tracking
				_satellite.track('add_to_basket');
			});      
		    return this;
	    },


	    /**
	     * Track confirmAndPay
	     */	    
	    _trackConfirmAndPay: function() {
			var prods, $dataAttributesSelector;
						
			if ( this.step.step === 'checkout/ordersummary' ) {
				prods = [];
				$dataAttributesSelector = $('tr.prod');

				$dataAttributesSelector.each(function(){
					prods.push( ';' + $(this).data('basket-itemsku') ); 
				});

				prods = prods.join(',');
				
				digitalData.prod = prods;
				digitalData.event_track = 'confirmPay';
			}
		    return this;
		},


	    /**
	     * Track order
	     */	    
	    _trackOrder: function() {
			var prods, prodsArray, ord, transaction, pay;
			
			if ( this.step.step ==='checkout/orderreceipt' ) {
				prodsArray = [];
				ord = Tracking.orderJSON;
				prods = Tracking.orditemsArray;
				
				$.each( prods, function( key, value ) {
					var sku 	= value.sku,
						price 	= value.price,
						qty 	= value.qty;
					
					prodsArray.push( ';' + sku + ';' + qty + ';' + ( price * qty ) );
				});
				
				prods = prodsArray.join(',');
				
				/* Translate venda_ohpaytype values into English*/
				pay = ord.payment;
        switch (ord.payment) {
          case '0':
            pay = 'Credit Card';
            break;
          case '5':
            pay = 'Manual Payment';
            break;
          case '8':
            pay = 'Free Order';
            break;
          case '12':
            pay = 'PayPal';
            break;
          default:
            pay = 'Payment Type' + pay;
        }
				
				/**
				 * Prevent duplicate orders using sessionStorage
				 * i.e. if the user refreshes the browser on order receipt
				 */
				if (Modernizr.sessionstorage) {
				  /* sessionStorage is available */
				  if (!sessionStorage.transactionId || sessionStorage.transactionId !== ord.ref) {
				    /* Only send order data if the order ref has changed */
				    sessionStorage.transactionId = ord.ref;
		  			digitalData.orderId = ord.ref;
		  			digitalData.orderTax = ord.tax;
		  			digitalData.orderShipping = ord.shipping;
		  			digitalData.orderDiscount = ord.discount;
		  			digitalData.orderTotal = ord.total;
		  			digitalData.paymentType = pay;
		  			digitalData.orderCurrency = ord.currency;
		  			digitalData.prod = prods;
		  			digitalData.event_track = 'purchase';
				  }
				} else {
				  /* sessionStorage not available, so send order data anyway*/
		  			digitalData.orderId = ord.ref;
		  			digitalData.orderTax = ord.tax;
		  			digitalData.orderShipping = ord.shipping;
		  			digitalData.orderDiscount = ord.discount;
		  			digitalData.orderTotal = ord.total;
		  			digitalData.paymentType = pay;
		  			digitalData.orderCurrency = ord.currency;
		  			digitalData.prod = prods;
		  			digitalData.event_track = 'purchase';	
				}	
			}			
			return this;
		},


	    /**
	     * Track pageType
	     * For non-workflow retreive value from Tracking.pageType();
	     * For workflow use Tracking.step();
	     */	    
	    _trackPageType: function() {
			var pageType;
			
			pageType = ( this.step ) ? this.chkStep.type : this.pageType;
			
			digitalData.pageType = pageType;
		    return this;
		},


	    /**
	     * Track pageName and channel 
	     */	    
	    _trackPageNameAndChannel: function() {
		    var pageName, channel, section, sub_section;

			/* Section and subsection should return blank values if no data avaiulable for them. Hence why declaring them here. */
			section = '';
			sub_section = '';
  
			if ( this.step ) {
				/* Workflow set pageName and Channel based on workflow and step name */
				pageName = step.step.replace('/',':');
				channel = $('#tag-workflow').text();
			} else {
				/* Non workflow */
				if ( this.pageType == 'Home' ) {
					/* Home */
					pageName = 'Home';
					channel = 'Home';
				} else if ( this.pageType == 'Content') {
					/* Static pages */
					pageName = 'Home';
					channel = 'Home';
				} else {
					/* icat and invt pages*/
					if ( window.location.pathname.indexOf('/shop&') > -1 ) {
						/* If on New In page hardcode the values as this is the 'shop' category shown in a PLP template */
						pageName = 'New In';
						channel = 'New In';
					} else {
						/* Any other invt or icat template */
						pageName = "";	
						/* Remove preceeding colon */
						pageName = pageName.substr(1);
						/* Remove trailing colon. Present on icat pages */
						pageName = ( this.pageType == 'Category' ) ? pageName.slice(0, -1) : pageName;
						channel = '';	
					

						/* Populate section and subsection with last two properties of pageName if pageName contains 2 or more properties in it's taxonomy */
						/* Create array from colon delimited pageName value */
						section = (pageName != '') ? pageName.split(':') : '';
						if ( section.length == 2 ) {
							section = section[section.length - 1];
						} else if ( section.length > 2 ) {
							sub_section = section[section.length -1];
							section = section[section.length - 2];
						}
					}					
				}
			}
			
			digitalData.channel = channel;
			digitalData.pageName = pageName;
			digitalData.section = section;
			digitalData.sub_section = sub_section;
		    return this;
	    },


	    /**
	     * Track session data
	     */
	    _trackVisitor: function () {
			var visitor = {
				status: (this.ses.ustype == 'Guest') ? 'Anonymous' : 'logged in'
			};
			
			digitalData.visitor = visitor;
			return this;
	    },
	    

	    /**
	     * Track PLP and search impressions / interations
	     */
	    _trackPLPSearch: function () {
			var catTemplate, searchTemplate, searchTerm, searchFilter, searchResults;
			
			catTemplate = $('div[lmzone=category]').length;
			searchTemplate = $('div[lmzone=Search]').length;
			
			/* Only fire on PLP and serach results pages */
			if ( catTemplate > 0 || searchTemplate > 0 ) {
				LM.complete(function(){
					searchTerm = $('.search-keyword-search-text').text() || $('#parentCatRef').text();
					searchResults = $('.search-total-results').text();
					/* If zero search results set searResults to be a sring; 'zero' */
					searchResults = (searchResults == 0) ? 'zero' : searchResults.toString();
					/* Create colon seperated string from applied filters */
					searchFilter = $('.js-removesearch.search-refine-link').map(function(){return this.text}).get().join(':');
					
		
					global.digitalData.searchTerm = searchTerm;
					global.digitalData.searchResults = searchResults;
					global.digitalData.searchFilter = searchFilter;
					//Fire tracking
					_satellite.track('search_interaction');
				});				
			}
			return this;
	    },


	    /**
	     * Track PDP impression
	     */
	    _trackPDPImpression: function () {
  			var prodName, prod_category;
  			
  			if ( this.pageType === 'Product' ) {
  				prodName = $('#tag-invtref').text();
  				prod_category =  'shop';
  				prod_category = prod_category.toLowerCase();
  				
  				digitalData.prod = ';'+prodName;
  				digitalData.prod_category = prod_category;
          // Prod view for non ticket-booking flow
          if ( window.location.hash !== '#book-now' ) {
            digitalData.event_track = 'prodview';
          }
  				return this;
  			}
	    },

	    /**
	     * Track Basket impressions
	     */
	    _trackBasketImpression: function () {
			var prods;
			
			if ( this.step.step ==='shopcart/shopcart' ) {
				prods = [];
				
				$('tr.prod').each(function(){
					prods.push( ';' + $(this).data('basket-itemsku') ); 				
				});
				
				prods = prods.join(',');
				
				digitalData.prod = prods;
				digitalData.event_track = 'scView';
				//Fire tracking
				_satellite.track('cart_view');				
			}			
	    }

    };

    return adobeTracking._init();
  }

  global.digitalData = defineModule(global.jQuery, global.Venda.Widget.Tracking);
  /* Fire Tracking */
  _satellite.pageBottom();
}(this);
